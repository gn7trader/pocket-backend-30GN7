import express from "express";
import cors from "cors";
import bodyParser from "body-parser";
import { io } from "socket.io-client";

const app = express();
app.use(cors());
app.use(bodyParser.json());

// URL do WebSocket da PocketOption
const WS_URL = "wss://api-in.pocketoption.com:8095/socket.io/?EIO=3&transport=websocket";

// Teste de conexão WebSocket
console.log("?? Tentando conectar ao WebSocket...");
const socket = io(WS_URL, {
  transports: ["websocket"],
  reconnectionAttempts: 3, // tenta 3 vezes
  timeout: 5000,           // tempo máximo 5s
});

socket.on("connect", () => {
  console.log("? Conectado ao WebSocket com sucesso!");
});

socket.on("connect_error", (err) => {
  console.error("? Erro ao conectar no WebSocket:", err.message);
});

socket.on("disconnect", () => {
  console.warn("?? WebSocket desconectado.");
});

// Rota teste
app.get("/", (req, res) => {
  res.send("?? Backend PocketOption rodando no Render!");
});

// Exemplo de endpoint saldo
app.post("/saldo", (req, res) => {
  const { ssid } = req.body;
  res.json({ saldo: 1000, ssidRecebido: ssid });
});

// Porta dinâmica do Render
app.listen(process.env.PORT || 8000, () => {
  console.log(`Servidor rodando na porta ${process.env.PORT || 8000}`);
});
